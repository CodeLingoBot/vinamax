Examples

First example:

package main

import (
        . "github.com/JLeliaert/vinamax"
        "log"
        "math"
)

func main() {
        //first define geometry

        World(0, 0, 0, 1e-6)
        test := Cube{S: 1e-8}
        test.Addparticles(500)
        Maketree()

        Particle_radius(7.0e-9)
        //        Lognormal_radius(1.6e-9)

        //An example of an arbitrary function for the external field (in Tesla, t in seconds)
        B_ext = func(t float64) (float64, float64, float64) { return 0.01, 0.001*math.Sin(t), t / 2 }

        Demag = false
        FMM = false
        Dt = 1e-11
        T = 0.
        Temp = 300.
        SetRandomSeed(3)
        Alpha = 0.01
        Ku1 = 10000
        Msat(400000)

        //Anisotropy_axis(0, 0, 1)
        Anisotropy_random()
        M_uniform(0, 0, 1)

        //Tableadd_B_eff_at_location(0,0.0,0)
        Tableadd("B_ext")
        Output(5e-10)
        Save("geometry")

        Run(1e-8)
}
